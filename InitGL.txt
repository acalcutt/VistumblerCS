    private void InitOpenGl()
    {
        if (_glReady) return;
        _hDC = GetDC(_childHwnd);
        if (_hDC == IntPtr.Zero) throw new Exception(""GetDC failed"");

        var pfd = new PIXELFORMATDESCRIPTOR();
        pfd.nSize = (ushort)Marshal.SizeOf<PIXELFORMATDESCRIPTOR>();
        pfd.nVersion = 1;
        pfd.dwFlags = PFD_DRAW_TO_WINDOW | PFD_SUPPORT_OPENGL | PFD_DOUBLEBUFFER;
        pfd.iPixelType = 0; // PFD_TYPE_RGBA
        pfd.cColorBits = 24;
        pfd.cDepthBits = 24;
        pfd.cStencilBits = 8;
        pfd.iLayerType = 0; // PFD_MAIN_PLANE

        int format = ChoosePixelFormat(_hDC, ref pfd);
        if (format == 0) throw new Exception(""ChoosePixelFormat failed"");
        if (!SetPixelFormat(_hDC, format, ref pfd)) throw new Exception(""SetPixelFormat failed"");

        IntPtr tempContext = wglCreateContext(_hDC);
        wglMakeCurrent(_hDC, tempContext);

        IntPtr pCreateContextAttribsARB = wglGetProcAddress(""wglCreateContextAttribsARB"");
        if (pCreateContextAttribsARB != IntPtr.Zero)
        {
            var createAttribs = Marshal.GetDelegateForFunctionPointer<WglCreateContextAttribsARBDelegate>(pCreateContextAttribsARB);
            var attribs = new[]
            {
                WGL_CONTEXT_MAJOR_VERSION_ARB, 3,
                WGL_CONTEXT_MINOR_VERSION_ARB, 3,
                WGL_CONTEXT_PROFILE_MASK_ARB, WGL_CONTEXT_COMPATIBILITY_PROFILE_BIT,
                0
            };
            _hGLRC = createAttribs(_hDC, IntPtr.Zero, attribs);
            wglMakeCurrent(IntPtr.Zero, IntPtr.Zero);
            wglDeleteContext(tempContext);
            wglMakeCurrent(_hDC, _hGLRC);
        }
        else
        {
            _hGLRC = tempContext;
        }

        IntPtr versionPtr = glGetString(GL_VERSION);
        string versionStr = Marshal.PtrToStringAnsi(versionPtr) ?? ""Unknown"";
        Debug.WriteLine($""[MapHost] OpenGL Version: {versionStr}"");
        System.IO.File.AppendAllText(""map_log.txt"", $""[MapHost] OpenGL Version: {versionStr}\n"");

        _glReady = true;
    }

    private void InitMaplibre()
    {
        float dpi = (float)(PresentationSource.FromVisual(this)?.CompositionTarget?.TransformToDevice.M11 ?? 1.0);
        int w = Math.Max(1, (int)ActualWidth);
        int h = Math.Max(1, (int)ActualHeight);

        System.IO.File.AppendAllText(""map_log.txt"", $""[MapHost] InitMaplibre: w={w} h={h} dpi={dpi}\n"");

        var sizeLogical = new MaplibreNative.Size((uint)w, (uint)h);
        var sizePhysical = new MaplibreNative.Size((uint)(w * dpi), (uint)(h * dpi));

        _runLoop = new RunLoop(RunLoop.Type.New);

        _frontend = new ExternalRenderingContextFrontend(_hDC, _hGLRC, sizePhysical, dpi);
        _frontend.Updated += (parameters) =>
        {
            if (_renderPending) return;
            _renderPending = true;
            Dispatcher.BeginInvoke(DispatcherPriority.Render, () =>
            {
                _renderPending = false;
                if (!_glReady || _frontend == null) return;
                wglMakeCurrent(_hDC, _hGLRC);
                glClearColor(0f, 0f, 0.45f, 1f);
                glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT | GL_STENCIL_BUFFER_BIT);
                _frontend.Render();
                uint glErr = glGetError();
                if (glErr != 0) Debug.WriteLine($""[MapHost] GL error after render: 0x{glErr:X}"");
                SwapBuffers(_hDC);
            });
        };

        var mapOptions = new MapOptions().WithMapMode(MapMode.Continuous).WithSize(sizeLogical).WithPixelRatio(dpi);
        var resOptions = new ResourceOptions()
            .WithCachePath(System.IO.Path.Combine(System.IO.Path.GetTempPath(), ""maplibre_cache.db""))
            .WithAssetPath(AppDomain.CurrentDomain.BaseDirectory);
        
        MapLibreLogger.SetObserver((severity, type, code, msg) => {
            System.IO.File.AppendAllText(""map_log.txt"", $""[MapLibre:{severity}:{type}] {msg}\n"");
        });

        var obs = new MapObserver();
        obs.DidFailLoadingMap += (mode, err) => System.IO.File.AppendAllText(""map_log.txt"", $""[MapHost] Error: {err}\n"");
        
        _map = new Map(_frontend, obs, mapOptions, resOptions);
        _map.Style.LoadURL(StyleUrl);
    }
