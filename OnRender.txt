    protected override void OnRenderSizeChanged(SizeChangedInfo info)
    {
        base.OnRenderSizeChanged(info);
        
        float dpi = (float)(PresentationSource.FromVisual(this)?.CompositionTarget?.TransformToDevice.M11 ?? 1.0);
        int w = Math.Max(1, (int)info.NewSize.Width);
        int h = Math.Max(1, (int)info.NewSize.Height);
        var sizeLogical = new MaplibreNative.Size((uint)w, (uint)h);
        var sizePhysical = new MaplibreNative.Size((uint)(w * dpi), (uint)(h * dpi));

        if (_childHwnd != IntPtr.Zero)
            SetWindowPos(_childHwnd, IntPtr.Zero, 0, 0, (int)sizePhysical.Width, (int)sizePhysical.Height, 0x0040);

        if (_frontend != null && w > 1 && h > 1)
        {
            if (_frontend.Backend.Size.Width != sizePhysical.Width || _frontend.Backend.Size.Height != sizePhysical.Height) {
                System.IO.File.AppendAllText(""map_log.txt"", $""[MapHost] Forcing SetSize to {sizePhysical.Width}x{sizePhysical.Height}\n"");
                _map?.SetSize(sizeLogical);
                _frontend.Backend.Size = sizePhysical;
            }
        }
    }
