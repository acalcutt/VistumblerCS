        long ticks = 0;
        _renderTimer = new DispatcherTimer(DispatcherPriority.Render)
        {
            Interval = TimeSpan.FromMilliseconds(16)
        };
        _renderTimer.Tick += (_, _) =>
        {
            ticks++;
            if (ticks % 100 == 0) System.IO.File.AppendAllText(""map_log.txt"", $""[MapHost] Timer tick {ticks}\n"");
            
            if (_frontend != null && ActualWidth > 1 && ActualHeight > 1) {
                float dpi = (float)(PresentationSource.FromVisual(this)?.CompositionTarget?.TransformToDevice.M11 ?? 1.0);
                var szLogical = new MaplibreNative.Size((uint)ActualWidth, (uint)ActualHeight);
                var szPhysical = new MaplibreNative.Size((uint)(ActualWidth * dpi), (uint)(ActualHeight * dpi));
                if (_frontend.Backend.Size.Width != szPhysical.Width || _frontend.Backend.Size.Height != szPhysical.Height) {
                    System.IO.File.AppendAllText(""map_log.txt"", $""[MapHost] Forcing SetSize to {szPhysical.Width}x{szPhysical.Height}\n"");
                    _map?.SetSize(szLogical);
                    _frontend.Backend.Size = szPhysical;
                    if (_childHwnd != IntPtr.Zero) SetWindowPos(_childHwnd, IntPtr.Zero, 0, 0, (int)szPhysical.Width, (int)szPhysical.Height, 0x0040);
                }
            }

            _runLoop?.RunOnce();
            RequestRender();
        };
    }
